## Overview

This document defines the `maxion-detection-types` crate, which consolidates all shared types used across the anti-cheat detection system. This eliminates duplication, ensures consistency, and provides a single source of truth for data structures.

//! Cryptographic types for signatures and tokens

use serde::{Deserialize, Serialize};
use ed25519_dalek::{Signer, SigningKey, VerifyingKey};
use blake3::Hash;

/// Ed25519 key pair for client authentication
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ClientKeypair {
    #[serde(with = "serde_bytes")]
    pub public_key: Vec<u8>,
    #[serde(with = "serde_bytes")]
    pub private_key: Vec<u8>,
}

impl ClientKeypair {
    /// Generate a new key pair
    pub fn generate() -> Self {
        let signing_key = SigningKey::generate(&mut rand::rngs::OsRng);
        let verifying_key = signing_key.verifying_key();
        
        Self {
            public_key: verifying_key.to_bytes().to_vec(),
            private_key: signing_key.to_bytes().to_vec(),
        }
    }
    
    /// Derive player ID from public key using BLAKE3
    pub fn derive_player_id(&self) -> String {
        let hash = blake3::hash(&self.public_key);
        hex::encode(hash.as_bytes())
    }
}

/// Request for an action token (one-time use authorization)
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ActionTokenRequest {
    pub player_id: String,
    pub nonce: String,
}

/// One-time action token generated by server
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ActionToken {
    pub player_id: String,
    pub timestamp: u64,
    pub nonce: String,
    #[serde(with = "serde_bytes")]
    pub token_hash: Vec<u8>,
    pub expires_at: u64,
}

impl ActionToken {
    /// Check if token is still valid
    pub fn is_valid(&self) -> bool {
        let now = chrono::Utc::now().timestamp() as u64;
        now < self.expires_at
    }
}

/// Payload signed by Ed25519
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SignedPayload {
    pub action_data: serde_json::Value,
    pub action_token: ActionToken,
}

/// Request with Ed25519 signature
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SignedRequest {
    pub payload: SignedPayload,
    #[serde(with = "serde_bytes")]
    pub signature: Vec<u8>,
}

impl SignedRequest {
    /// Verify Ed25519 signature
    pub fn verify(&self, public_key: &[u8]) -> Result<bool, CryptoError> {
        use ed25519_dalek::{Verifier, VerifyingKey};
        
        let payload_bytes = bincode::serialize(&self.payload)
            .map_err(|e| CryptoError::SerializationError(e.to_string()))?;
        
        let signature = ed25519_dalek::Signature::try_from(self.signature.as_slice())
            .map_err(|_| CryptoError::InvalidSignature)?;
        
        let verifying_key = VerifyingKey::from_bytes(public_key.try_into().unwrap())
            .map_err(|_| CryptoError::InvalidPublicKey)?;
        
        Ok(verifying_key.verify(&payload_bytes, &signature).is_ok())
    }
}

/// Cryptographic errors
#[derive(Debug, thiserror::Error)]
pub enum CryptoError {
    #[error("Serialization error: {0}")]
    SerializationError(String),
    #[error("Invalid signature")]
    InvalidSignature,
    #[error("Invalid public key")]
    InvalidPublicKey,
    #[error("Token expired")]
    TokenExpired,
    #[error("Invalid token hash")]
    InvalidTokenHash,
}

//! Detection event types

use serde::{Deserialize, Serialize};
use chrono::{DateTime, Utc};
use strum::{Display, EnumString};

/// Cheat event reported by client
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CheatEvent {
    pub game_id: String,
    pub version: String,
    pub player_id: String,
    pub cheat_type: CheatType,
    pub hwid: String,
    pub timestamp: DateTime<Utc>,
    pub detection_count: u32,
}

/// Types of cheats detected
#[derive(Debug, Clone, Serialize, Deserialize, Display, EnumString)]
pub enum CheatType {
    MemoryTampering,
    ValueFreeze,
    IntegrityViolation,
    SpeedHack,
    MacroDetection,
    ProcessInjection,
    DebuggerAttached,
    VmDetected,
    Unknown,
}

impl CheatType {
    /// Get severity level for a cheat type
    pub fn severity(&self) -> DetectionSeverity {
        match self {
            CheatType::DebuggerAttached => DetectionSeverity::High,
            CheatType::ProcessInjection => DetectionSeverity::High,
            CheatType::MemoryTampering => DetectionSeverity::Critical,
            CheatType::ValueFreeze => DetectionSeverity::Medium,
            CheatType::IntegrityViolation => DetectionSeverity::High,
            CheatType::SpeedHack => DetectionSeverity::High,
            CheatType::MacroDetection => DetectionSeverity::Low,
            CheatType::VmDetected => DetectionSeverity::Medium,
            CheatType::Unknown => DetectionSeverity::Low,
        }
    }
}

/// Severity levels for detection events
#[derive(Debug, Clone, Serialize, Deserialize, Display, Eq, PartialEq)]
pub enum DetectionSeverity {
    Low,
    Medium,
    High,
    Critical,
}

/// Detailed detection event from client

pub struct DetectionEvent {
    pub event_type: DetectionEventType,
    pub severity: DetectionSeverity,

    ApiBypass,

    pub events: Vec<DetectionEvent>,
